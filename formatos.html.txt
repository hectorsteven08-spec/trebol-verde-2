<script>
  const buscador = document.getElementById("buscador");
  const enlaces = document.querySelectorAll("ul li");
  const sinResultados = document.querySelector(".no-resultados");

  buscador.addEventListener("keyup", () => {
    const texto = buscador.value.toLowerCase();
    let encontrados = 0;

    enlaces.forEach(li => {
      const textoEnlace = li.textContent.toLowerCase();
      if (textoEnlace.includes(texto)) {
        li.style.display = "";
        encontrados++;
      } else {
        li.style.display = "none";
      }
    });

    sinResultados.style.display = encontrados === 0 && texto.length > 0 ? "block" : "none";
  });
</script>
